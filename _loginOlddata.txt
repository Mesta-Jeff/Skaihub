export default function Login({ navigation }) {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [loading, setLoading] = useState(false);
  const [imageSrc, setImageSrc] = useState(null);

  const loginUser = () => {
    return fetch('http://192.168.89.13:8000/api/v2/authentication/login', {
      method: 'POST',
      headers: {
        'Accept': 'application/json',
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        email: email,
        password: password,
      }),
    })
    .then((response) => response.json())
    .then((responseJson) => {
      setLoading(false);
      if (responseJson.success) {
        setImageSrc(responseJson.data.image); // Set the image source
        Alert.alert(
          'Success Alert',
          `User found successfully. Token is: ${responseJson.data.token}`,
          [{ text: 'OK', onPress: () => navigation.push('Home') }]
        );
      } else {
        Alert.alert('Error Alert', responseJson.message || 'Failed to authenticate user');
      }
    })
    .catch((error) => {
      setLoading(false);
      Alert.alert('Error', `An error occurred. Please try again. ${error}`);
      console.error('Error:', error);
    });
  };

  const handlePress = () => {
    if (!email || !password) {
      Alert.alert('Skai Alert', 'Sorry, an error has been found. All fields are required.');
      return;
    }

    setLoading(true);
    loginUser();
  };
